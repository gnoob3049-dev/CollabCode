   180→      // Persist to room via PATCH
   181→      fetch(`/api/rooms/${currentRoomId}`, {
   182→        method: 'PUT',
   183→        headers: { 'Content-Type': 'application/json' },
   184→        credentials: 'include',

   186→          activityLog: [...(currentRoom?.activityLog || []), entry],

   196→  useEffect(() => {
   197→    if (!currentRoomId || !user) return;

   199→    const ydoc = new Y.Doc();
   200→    ydocRef.current = ydoc;
   201→

   203→    fileListRef.current = fileList;
   204→
   205→    // Connect to y-websocket provider
   206→    const wsProtocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
   207→    const wsUrl = `${wsProtocol}//${window.location.host}/yjs?XTransformPort=3003`;
   208→    const provider = new WebsocketProvider(wsUrl, currentRoomId, ydoc, {
   209→      connect: true,
   210→    });
   211→    providerRef.current = provider;
   212→
   213→    // Fetch room data
   214→    fetch(`/api/rooms/${currentRoomId}`, {
   215→      credentials: 'include',
   216→    })
   217→      .then((res) => res.json())
   218→      .then((data) => {
   219→        if (data.room) {
   220→          setCurrentRoom(data.room);
   221→          setLanguage(data.room.language || 'javascript');
   222→          if (data.room.activityLog) {
   223→            setActivities(data.room.activityLog);
   224→          }
   225→          const roomFiles: string[] = data.room.files?.map(
   226→            (f: { name: string }) => f.name

   232→                if (!fileList.toArray().includes(name)) {

   237→                (f: { name: string; content: string }) => {

   240→                    ytext.insert(0, f.content);

   246→
   247→          // Set first file as active

   249→            setCurrentFileName(roomFiles[0]);
   250→            initDoneRef.current = true;
   251→          }
   252→        }
   253→      })
   254→      .catch((err) => {
   255→        console.error('Failed to fetch room:', err);
   256→        toast.error('Failed to load room');
   257→      });
   258→
   259→    // Sync file list from Y.js
   260→    const updateFiles = () => {
   261→      const arr = fileList.toArray();

   263→    };
   264→    fileList.observe(updateFiles);
   265→    updateFiles();
   266→
   267→    // Set up UndoManager for version history tracking
   268→    const undoManager = new Y.UndoManager(ydoc.getText(currentFileName), {
   269→      trackedOrigins: new Set(['local']),
   270→    });
   271→    undoManagerRef.current = undoManager;
   272→
   273→    const createSnapshot = (type: 'edit' | 'add' | 'delete') => {
   274→      const ytext = ydoc.getText(currentFileName);
   275→      const content = ytext.toString();
   276→      // Skip if content hasn't changed since last snapshot
   277→      if (content === lastSnapshotContentRef.current) return;
   278→      lastSnapshotContentRef.current = content;
   279→
   280→      const lines = content.split('\n');
   281→      const snapshot: VersionSnapshot = {
   282→        id: `snap-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`,
   283→        timestamp: Date.now(),
   284→        content,
   285→        type,
   286→        lineCount: lines.length,
   287→        charCount: content.length,
   288→        fileName: currentFileName,
   289→        prevContent: undefined,
   290→      };
   291→
   292→      setVersionSnapshots((prev) => {
   293→        // Update prevContent on the new snapshot to point to the last one
   294→        const lastSnap = prev.length > 0 ? prev[prev.length - 1] : null;
   295→        if (lastSnap && lastSnap.fileName === currentFileName) {
   296→          snapshot.prevContent = lastSnap.content;
   297→        }
   298→        const updated = [...prev, snapshot];
   299→        // Limit to 50 snapshots
   300→        if (updated.length > 50) {
   301→          return updated.slice(updated.length - 50);
   302→        }
   303→        return updated;
   304→      });
   305→    };
   306→
   307→    // Debounced snapshot capture on undoable changes
   308→    const handleUndoManagerChange = () => {
   309→      if (debounceTimerRef.current) {
   310→        clearTimeout(debounceTimerRef.current);
   311→      }

   313→        createSnapshot('edit');
   314→      }, 3000);
   315→    };
   316→
   317→    undoManager.on('stack-item-added', handleUndoManagerChange);
   318→
   319→    // Initial snapshot
   320→    setTimeout(() => {
   321→      const ytext = ydoc.getText(currentFileName);
   322→      const content = ytext.toString();

   324→        lastSnapshotContentRef.current = content;
   325→        setVersionSnapshots([{
   326→          id: `snap-init-${Date.now()}`,
   327→          timestamp: Date.now(),
   328→          content,
   329→          type: 'add',
   330→          lineCount: content.split('\n').length,
   331→          charCount: content.length,
   332→          fileName: currentFileName,
   333→        }]);
   334→      }
   335→    }, 1000);

   426→
   427→  // Bind Monaco to Y.Text when file changes
   428→  const handleEditorMount: OnMount = useCallback(
   429→    (editorInstance, monaco) => {

   492→        ytext,
   493→        model,
   494→        new Set([editorInstance]),
   495→        providerRef.current.awareness

   505→        return;
   506→      }

   509→      if (!list.toArray().includes(name)) {
   510→        list.push([name]);
   511→        // Initialize empty content if not exists
   512→        const ytext = ydocRef.current.getText(name);
   513→        if (!ytext.toString()) {
   514→          ytext.insert(0, '');
   515→        }
   516→        setCurrentFileName(name);
   517→        logActivity('file_add', `added file ${name}`);
   518→      }
   519→    },
   520→    [isLockedForUser, logActivity]
   521→  );
   522→
   523→  const handleRenameFile = useCallback(
   524→    (oldName: string, newName: string) => {
   525→      if (!fileListRef.current || !ydocRef.current) return;
   526→      const list = fileListRef.current;
   527→      const idx = list.toArray().indexOf(oldName);
   528→      if (idx === -1) return;
   529→
   530→      // Copy content to new name
   531→      const ydoc = ydocRef.current;
   532→      const oldContent = ydoc.getText(oldName).toString();
   533→      ydoc.getText(newName).insert(0, oldContent);
   534→      ydoc.getText(oldName).delete(0, oldContent.length);
   535→
   536→      // Update file list
   537→      list.delete(idx, 1);
   538→      list.insert(idx, [newName]);
   539→
   540→      // Update current file if renamed
   541→      if (currentFileName === oldName) {
   542→        setCurrentFileName(newName);
   543→      }
   544→    },
   545→    [currentFileName, setCurrentFileName]
   546→  );
   547→
   548→  const handleDeleteFile = useCallback(
   549→    (name: string) => {
   550→      if (isLockedForUser) {
   551→        toast.error('This room is in read-only mode');
   552→        return;
   553→      }
   554→      if (!fileListRef.current || !ydocRef.current) return;
   555→      const list = fileListRef.current;
   556→      const idx = list.toArray().indexOf(name);
   557→      if (idx === -1) return;
   558→
   559→      list.delete(idx, 1);
   560→
   561→      // Delete the Y.Text content
   562→      const ytext = ydocRef.current.getText(name);
   563→      const content = ytext.toString();
   564→      ytext.delete(0, content.length);
   565→
   566→      // Select another file
   567→      if (currentFileName === name) {
   568→        const remaining = list.toArray();
   569→        if (remaining.length > 0) {
   570→          setCurrentFileName(remaining[0]);
   571→        } else {
   572→          // Create a default file
   573→          handleCreateFile('index.js');
   574→        }
   575→      }
   576→      logActivity('file_delete', `deleted file ${name}`);
   577→    },
   578→    [currentFileName, setCurrentFileName, handleCreateFile, isLockedForUser, logActivity]
   579→  );
   580→
   581→  // Run code
   582→  const handleRun = useCallback(async () => {
   583→    if (isLockedForUser) {
   584→      toast.error('This room is in read-only mode');
   585→      return;
   586→    }
   587→    if (!ydocRef.current) return;
   588→    setIsRunning(true);
   589→    setOutput('');
   590→    setOutputPanelOpen(true);
   591→    playSound('run_start');
   592→    try {
   593→      const ytext = ydocRef.current.getText(currentFileName);
   594→      const code = ytext?.toString() || '';
   595→      const fileLanguage = getLanguageFromFilename(currentFileName);
   596→
   597→      const res = await fetch('/api/run', {
   598→        method: 'POST',
   599→        headers: { 'Content-Type': 'application/json' },
   600→        credentials: 'include',
   601→        body: JSON.stringify({ code, language: fileLanguage }),
   602→      });
   603→      const data = await res.json();
   604→      const hasErrors = !!data.error;
   605→      setOutput(data.output || data.error || 'No output');
   606→      playSound('run_complete', { hasErrors });
   607→      if (hasErrors) {
   608→        const firstLine = (data.error || 'Unknown error').split('\n')[0];
   609→        toast.error(firstLine.length > 80 ? firstLine.slice(0, 80) + '…' : firstLine);
   610→      } else {
   611→        toast.success('Code executed successfully');
   612→      }
