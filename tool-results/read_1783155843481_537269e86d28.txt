   278→
   279→    // Set up UndoManager for version history tracking
   280→    const undoManager = new Y.UndoManager(ydoc.getText(currentFileName), {

   284→
   285→    const createSnapshot = (type: 'edit' | 'add' | 'delete') => {
   286→      const ytext = ydoc.getText(currentFileName);
   287→      const content = ytext.toString();
   288→      // Skip if content hasn't changed since last snapshot
   289→      if (content === lastSnapshotContentRef.current) return;
   290→      lastSnapshotContentRef.current = content;
   291→
   292→      const lines = content.split('\n');
   293→      const snapshot: VersionSnapshot = {
   294→        id: `snap-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`,
   295→        timestamp: Date.now(),
   296→        content,
   297→        type,
   298→        lineCount: lines.length,
   299→        charCount: content.length,
   300→        fileName: currentFileName,
   301→        prevContent: undefined,
   302→      };
   303→
   304→      setVersionSnapshots((prev) => {
   305→        // Update prevContent on the new snapshot to point to the last one
   306→        const lastSnap = prev.length > 0 ? prev[prev.length - 1] : null;
   307→        if (lastSnap && lastSnap.fileName === currentFileName) {
   308→          snapshot.prevContent = lastSnap.content;
   309→        }
   310→        const updated = [...prev, snapshot];
   311→        // Limit to 50 snapshots
   312→        if (updated.length > 50) {
   313→          return updated.slice(updated.length - 50);
   314→        }
   315→        return updated;
   316→      });
   317→    };
   318→

   322→        clearTimeout(debounceTimerRef.current);

   325→        createSnapshot('edit');
   326→      }, 3000);
   327→    };

   330→

   341→          type: 'add',

   515→  // File operations

   519→        toast.error('This room is in read-only mode');
   520→        return;
   521→      }

   523→      const list = fileListRef.current;
   524→      if (!list.toArray().includes(name)) {
   525→        list.push([name]);

   527→        const ytext = ydocRef.current.getText(name);
   528→        if (!ytext.toString()) {
   529→          ytext.insert(0, '');
   530→        }
   531→        setCurrentFileName(name);
   532→        logActivity('file_add', `added file ${name}`);
   533→      }
   534→    },
   535→    [isLockedForUser, logActivity]
   536→  );
   537→
   538→  const handleRenameFile = useCallback(
   539→    (oldName: string, newName: string) => {
   540→      if (!fileListRef.current || !ydocRef.current) return;
   541→      const list = fileListRef.current;
   542→      const idx = list.toArray().indexOf(oldName);
   543→      if (idx === -1) return;
   544→
   545→      // Copy content to new name
   546→      const ydoc = ydocRef.current;
   547→      const oldContent = ydoc.getText(oldName).toString();
   548→      ydoc.getText(newName).insert(0, oldContent);
   549→      ydoc.getText(oldName).delete(0, oldContent.length);
   550→
   551→      // Update file list
   552→      list.delete(idx, 1);
   553→      list.insert(idx, [newName]);
   554→
   555→      // Update current file if renamed
   556→      if (currentFileName === oldName) {
   557→        setCurrentFileName(newName);
   558→      }
   559→    },
   560→    [currentFileName, setCurrentFileName]
   561→  );
   562→
   563→  const handleDeleteFile = useCallback(
   564→    (name: string) => {
   565→      if (isLockedForUser) {
   566→        toast.error('This room is in read-only mode');
   567→        return;
   568→      }
   569→      if (!fileListRef.current || !ydocRef.current) return;
   570→      const list = fileListRef.current;
   571→      const idx = list.toArray().indexOf(name);
   572→      if (idx === -1) return;
   573→
   574→      list.delete(idx, 1);
   575→
   576→      // Delete the Y.Text content
   577→      const ytext = ydocRef.current.getText(name);
   578→      const content = ytext.toString();
   579→      ytext.delete(0, content.length);
   580→
   581→      // Select another file
   582→      if (currentFileName === name) {
   583→        const remaining = list.toArray();
   584→        if (remaining.length > 0) {
   585→          setCurrentFileName(remaining[0]);
   586→        } else {
   587→          // Create a default file
   588→          handleCreateFile('index.js');
   589→        }
   590→      }
   591→      logActivity('file_delete', `deleted file ${name}`);
   592→    },
   593→    [currentFileName, setCurrentFileName, handleCreateFile, isLockedForUser, logActivity]
   594→  );
   595→
   596→  // Run code
   597→  const handleRun = useCallback(async () => {
   598→    if (isLockedForUser) {
   599→      toast.error('This room is in read-only mode');
   600→      return;
   601→    }
   602→    if (!ydocRef.current) return;
   603→    setIsRunning(true);
   604→    setOutput('');
   605→    setRunResultIsHtml(false);
   606→    setRunResultIsCss(false);
   607→    setOutputPanelOpen(true);
   608→    playSound('run_start');
