   288→    const createSnapshot = (type: 'edit' | 'add' | 'delete') => {

   298→        timestamp: Date.now(),
   299→        content,
   300→        type,

   307→      setVersionSnapshots((prev) => {
   308→        // Update prevContent on the new snapshot to point to the last one

   310→        if (lastSnap && lastSnap.fileName === currentFileName) {
   311→          snapshot.prevContent = lastSnap.content;
   312→        }

   316→          return updated.slice(updated.length - 50);
   317→        }
   318→        return updated;
   319→      });

   551→      ydoc.getText(newName).insert(0, oldContent);
   552→      ydoc.getText(oldName).delete(0, oldContent.length);
   553→
   554→      // Update file list
   555→      list.delete(idx, 1);
   556→      list.insert(idx, [newName]);
   557→
   558→      // Update current file if renamed
   559→      if (currentFileName === oldName) {
   560→        setCurrentFileName(newName);
   561→      }
   562→    },

   575→        list.delete(0, list.length);
   576→        list.insert(0, newOrder);
   577→      });
   578→      logActivity('file_reorder', 'reordered files in the file tree');
   579→    },
   580→    [logActivity]
   581→  );

   583→  const handleDeleteFile = useCallback(
   584→    (name: string) => {
   585→      if (isLockedForUser) {
   586→        toast.error('This room is in read-only mode');
   587→        return;
   588→      }

   590→      const list = fileListRef.current;
   591→      const idx = list.toArray().indexOf(name);
   592→      if (idx === -1) return;
   593→
   594→      list.delete(idx, 1);
   595→
   596→      // Delete the Y.Text content
   597→      const ytext = ydocRef.current.getText(name);
   598→      const content = ytext.toString();
   599→      ytext.delete(0, content.length);
   600→
   601→      // Select another file
   602→      if (currentFileName === name) {
   603→        const remaining = list.toArray();
   604→        if (remaining.length > 0) {
   605→          setCurrentFileName(remaining[0]);
   606→        } else {
