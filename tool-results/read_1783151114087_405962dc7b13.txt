   104→   104→function timeAgo(dateStr?: string): string {
   105→   105→  if (!dateStr) return "Unknown";
   106→   106→  const now = Date.now();
   107→   107→  const then = new Date(dateStr).getTime();
   108→   108→  const diff = now - then;
   109→   109→  const minutes = Math.floor(diff / 60000);
   110→   110→  if (minutes < 1) return "Just now";
   111→   111→  if (minutes < 60) return `${minutes}m ago`;
   112→   112→  const hours = Math.floor(minutes / 60);
   113→   113→  if (hours < 24) return `${hours}h ago`;
   114→   114→  const days = Math.floor(hours / 24);
   115→   115→  return `${days}d ago`;
   116→   116→}
   117→   117→
   118→   118→function isRecent(dateStr?: string): boolean {
   119→   119→  if (!dateStr) return false;
   120→   120→  const diff = Date.now() - new Date(dateStr).getTime();
   121→   121→  return diff < 30 * 60 * 1000; // within 30 minutes
   122→   122→}
