   104→function timeAgo(dateStr?: string): string {
   105→  if (!dateStr) return "Unknown";
   106→  const now = Date.now();
   107→  const then = new Date(dateStr).getTime();
   108→  const diff = now - then;
   109→  const minutes = Math.floor(diff / 60000);
   110→  if (minutes < 1) return "Just now";
   111→  if (minutes < 60) return `${minutes}m ago`;
   112→  const hours = Math.floor(minutes / 60);
   113→  if (hours < 24) return `${hours}h ago`;
   114→  const days = Math.floor(hours / 24);
   115→  return `${days}d ago`;
   116→}
   117→
   118→function isRecent(dateStr?: string): boolean {
   119→  if (!dateStr) return false;
   120→  const diff = Date.now() - new Date(dateStr).getTime();
   121→  return diff < 30 * 60 * 1000; // within 30 minutes
   122→}
