   289→    const createSnapshot = (type: 'edit' | 'add' | 'delete') => {
   290→      const ytext = ydoc.getText(currentFileName);
   291→      const content = ytext.toString();

   293→      if (content === lastSnapshotContentRef.current) return;

   295→
   296→      const lines = content.split('\n');
   297→      const snapshot: VersionSnapshot = {
   298→        id: `snap-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`,
   299→        timestamp: Date.now(),
   300→        content,
   301→        type,
   302→        lineCount: lines.length,
   303→        charCount: content.length,
   304→        fileName: currentFileName,
   305→        prevContent: undefined,
   306→      };
   307→
   308→      setVersionSnapshots((prev) => {
   309→        // Update prevContent on the new snapshot to point to the last one
   310→        const lastSnap = prev.length > 0 ? prev[prev.length - 1] : null;
   311→        if (lastSnap && lastSnap.fileName === currentFileName) {
   312→          snapshot.prevContent = lastSnap.content;
   313→        }
   314→        const updated = [...prev, snapshot];
   315→        // Limit to 50 snapshots
   316→        if (updated.length > 50) {
   317→          return updated.slice(updated.length - 50);
   318→        }
   319→        return updated;
   320→      });
   321→    };

   542→  const handleRenameFile = useCallback(
   543→    (oldName: string, newName: string) => {
   544→      if (!fileListRef.current || !ydocRef.current) return;

   547→      if (idx === -1) return;

   549→      // Copy content to new name

   551→      const oldContent = ydoc.getText(oldName).toString();
   552→      ydoc.getText(newName).insert(0, oldContent);
   553→      ydoc.getText(oldName).delete(0, oldContent.length);
   554→
   555→      // Update file list
   556→      list.delete(idx, 1);
   557→      list.insert(idx, [newName]);
   558→
   559→      // Update current file if renamed
   560→      if (currentFileName === oldName) {
   561→        setCurrentFileName(newName);
   562→      }
   563→    },
   564→    [currentFileName, setCurrentFileName]
   565→  );

   567→  // Reorder files in Y.js

   569→    (newOrder: string[]) => {

   572→      const current = list.toArray();
   573→      // Only update if order actually changed
   574→      if (JSON.stringify(current) === JSON.stringify(newOrder)) return;
   575→      ydocRef.current.transact(() => {
   576→        list.delete(0, list.length);
   577→        list.insert(0, newOrder);
   578→      });
   579→      logActivity('file_reorder', 'reordered files in the file tree');
   580→    },
   581→    [logActivity]
   582→  );
   583→
   584→  const handleDeleteFile = useCallback(
   585→    (name: string) => {
   586→      if (isLockedForUser) {
   587→        toast.error('This room is in read-only mode');
   588→        return;
   589→      }
   590→      if (!fileListRef.current || !ydocRef.current) return;
   591→      const list = fileListRef.current;
   592→      const idx = list.toArray().indexOf(name);
   593→      if (idx === -1) return;
   594→
   595→      list.delete(idx, 1);
   596→
   597→      // Delete the Y.Text content
   598→      const ytext = ydocRef.current.getText(name);
   599→      const content = ytext.toString();
   600→      ytext.delete(0, content.length);
   601→
   602→      // Select another file
   603→      if (currentFileName === name) {
   604→        const remaining = list.toArray();
   605→        if (remaining.length > 0) {
   606→          setCurrentFileName(remaining[0]);
